25 template <class T, unsigned int VDimension>
26 void PSMMixedEffectsShapeMatrixAttribute<T,VDimension>
27 ::UpdateMeanMatrix()
28 {
29  vnl_vector<double> tempvect;
30  tempvect.set_size(this->GetMeanMatrix().rows());
31  const vnl_vector<double> expl = this->GetExplanatory();
32 
33  // For each sample, which are in the columns, ...
34  for (unsigned int i = 0; i < this->GetMeanMatrix().cols(); i++)
35  {
36  int group_indx = i / m_TimePointsPerIndividual;
37  tempvect = this->GetIntercept() + this->GetSlope() * expl(i);
38  tempvect = tempvect + m_InterceptRand.get_row(group_indx);
39  tempvect = tempvect + m_SlopeRand.get_row(group_indx) * expl(i);
40  // ... compute the mean.
41  this->GetMeanMatrix().set_column(i, tempvect);
42  }
43 }

45 template <class T, unsigned int VDimension>
46 void PSMMixedEffectsShapeMatrixAttribute<T,VDimension>::
47 EstimateParameters()
48 {
49  vnl_matrix<double> X = *this + this->GetMeanMatrix();
50  const vnl_vector<double> expl = this->GetExplanatory();
51 
52  // Number of samples
53  int num_shapes = static_cast<double>(X.cols());
54  this->m_NumberOfIndividuals = num_shapes / (this->GetDomainsPerShape()*this->GetTimePointsPerIndividual());
55  int nr = X.rows(); //number of points*3
56 
57  //set the sizes of random slope and intercept matrix
58  m_SlopeRand.set_size(m_NumberOfIndividuals, nr); //num_groups X num_points*3
59  m_InterceptRand.set_size(m_NumberOfIndividuals, nr); //num_groups X num_points*3
60 
61  vnl_matrix<double> fixed; //slopes + intercepts for all points
62  vnl_matrix<double> random; //slopes + intercepts for all groups, for all points
63  fixed.set_size(2, nr);
64  random.set_size(2, nr*m_NumberOfIndividuals);
65  vnl_matrix<double> Ds(2,2); //covariance matrix of random parameters (2x2)
66  Ds.set_identity(); //initialize to identity
67  double sigma2s = 1; //variance of error
68  vnl_matrix<double> identity_n;
69  identity_n.set_size(m_TimePointsPerIndividual, m_TimePointsPerIndividual);
70  identity_n.set_identity();
71  vnl_matrix<double> identity_2;
72  identity_2.set_size(2,2);
73  identity_2.set_identity();
74  vnl_matrix<double> *Ws=NULL, *Vs=NULL;
75  Ws = new vnl_matrix<double>[m_NumberOfIndividuals];
76  Vs = new vnl_matrix<double>[m_NumberOfIndividuals];
77  for (int i = 0; i < m_NumberOfIndividuals; i++)
78  {
79  Vs[i].set_size(m_TimePointsPerIndividual, m_TimePointsPerIndividual);
80  Ws[i].set_size(m_TimePointsPerIndividual, m_TimePointsPerIndividual);
81  }
82 
83  vnl_matrix<double> sum_mat1(2,2,0);
84  vnl_vector<double> sum_mat2(2);
85  sum_mat2.fill(0.0);
86 
87  vnl_vector<double> residual;
88  residual.set_size(m_TimePointsPerIndividual);
89  residual.fill(0.0);
90 
91  double ecorr = 0.0;
92  double tracevar = 0.0;
93 
94  vnl_matrix<double> bscorr(2,2,0.0);
95  vnl_matrix<double> bsvar(2,2,0.0);
96  vnl_matrix<double> Xp;
97  Xp.set_size(m_TimePointsPerIndividual, 2);
98  vnl_vector<double> y;
99  y.set_size(m_TimePointsPerIndividual);
100  vnl_vector<double> tempvect;
101  tempvect.set_size(2);
102 
103  for (int i = 0; i < nr; i++) //for all points (x,y,z coordinates)
104  {
105  sigma2s = 1.0;
106  Ds.set_identity();
107  for (int j = 0; j < 50; j++) //EM iterations
108  {
109  sum_mat1.fill(0.0);
110  sum_mat2.fill(0.0);
111  residual.fill(0.0);
112  ecorr = 0.0;
113  tracevar = 0.0;
114  bscorr.fill(0.0);
115  bsvar.fill(0.0);
116 
117  for (int k = 0; k < m_NumberOfIndividuals; k++)
118  {
119  for (int l = 0; l < m_TimePointsPerIndividual; l++)
120  {
121  Xp(l,0) = expl(k*m_TimePointsPerIndividual + l);
122  Xp(l,1) = 1;
123  y(l) = X(i, k*m_TimePointsPerIndividual + l);
124  }
125  Vs[k] = (identity_n * sigma2s) + Xp * Ds * vnl_transpose(Xp);
126  //Ws = static_cast<vnl_matrix> (vnl_matrix_inverse<double>(Vs));
127  Ws[k] = vnl_inverse(Vs[k]);
128  sum_mat1 = sum_mat1 + vnl_transpose(Xp) * Ws[k] * Xp;
129  sum_mat2 = sum_mat2 + vnl_transpose(Xp) * Ws[k] * y;
130  }
131  tempvect = vnl_inverse(sum_mat1) * sum_mat2;
132  fixed.set_column(i, tempvect);
133  for (int k = 0; k < m_NumberOfIndividuals; k++)
134  {
135  for (int l = 0; l < m_TimePointsPerIndividual; l++)
136  {
137  Xp(l,0) = expl(k*m_TimePointsPerIndividual + l);
138  Xp(l,1) = 1;
139  y(l) = X(i, k*m_TimePointsPerIndividual + l);
140  }
141  tempvect = Ds * vnl_transpose(Xp) * Ws[k] * (y - (Xp * fixed.get_column(i)));
142  random.set_column(i*m_NumberOfIndividuals + k, tempvect);
143  residual = y - (Xp * fixed.get_column(i)) - (Xp * random.get_column(i*m_NumberOfIndividuals + k));
144  ecorr = ecorr + dot_product(residual, residual);
145  tracevar = tracevar + (m_TimePointsPerIndividual - sigma2s * vnl_trace(Ws[k]));
146  bscorr = bscorr + outer_product(random.get_column(i*m_NumberOfIndividuals + k), random.get_column(i*m_NumberOfIndividuals + k));
147  bsvar = bsvar + (identity_2 - (vnl_transpose(Xp) * Ws[k] * Xp * Ds));
148  }
149  sigma2s = (ecorr + sigma2s * tracevar) / (num_shapes);
150  Ds = (bscorr + Ds * bsvar) / m_NumberOfIndividuals;
151  }//endfor EM iterations
152  //printf ("point #%d\n", i);
153  }//endfor all points on shape (x,y & z)
154 
155  this->SetSlope(fixed.get_row(0));
156  this->SetIntercept(fixed.get_row(1));
157  for (int i = 0; i < m_NumberOfIndividuals; i++)
158  {
159  for (int j = 0; j < nr; j++) //for all points * 3
160  {
161  m_SlopeRand(i,j) = random(0, j*m_NumberOfIndividuals+i);
162  m_InterceptRand(i,j) = random(1, j*m_NumberOfIndividuals+i);
163  }
164  }
165  delete [] Vs;
166  delete [] Ws;
167 
168  //printf ("points:\n");
169  //for (int k = 0; k < m_NumberOfIndividuals; k++)
170  // for (int l = 0; l < m_TimePointsPerIndividual; l++)
171  // printf ("%g %g\n", X(0,k*m_TimePointsPerIndividual + l), expl(k*m_TimePointsPerIndividual + l));
172 
173  //printf ("fixed: slope %g, intercept %g", m_Slope(0), this->GetIntercept()(0));
174  //printf ("random: slopes %g %g, intercepts %g %g", m_SlopeRand(0,0), m_SlopeRand(1,0), m_InterceptRand(0,0), m_InterceptRand(1,0));
175 }
