28 template <class TImage, class TShapeMatrix>
29 PSMEntropyModelFilter<TImage, TShapeMatrix>::PSMEntropyModelFilter()
30 {
31  // Default parameters for the multiscale functionality
32  m_CurrentScale = 0;
33  this->SetNumberOfScales(1);
34 
35  m_Initializing = false;
36 
37  m_ParticleSystem = PSMParticleSystem<Dimension>::New();
38 
39  // Here we create the various necessary components for the Shape
40  // Model optimization. First we construct the Shape Matrix.
41  m_ShapeMatrix = ShapeMatrixType::New();
42  m_ShapeMatrix->SetDomainsPerShape(1);
43 
44  // Now we create the Particle Entropy Function. This is the entropy
45  // estimation of particle position distributions on shape surfaces.
46  m_ParticleEntropyFunction
47  = PSMParticleEntropyFunction<typename ImageType::PixelType, Dimension>::New();
48 
49  // Now initialize the Shape Space entropy function and point it to
50  // the Shape Matrix attribute. The Shape Space entropy function
51  // computes entropy of the shape samples in n-dimensional shape
52  // space.
53  m_ShapeEntropyFunction = PSMShapeEntropyFunction<Dimension>::New();
54  m_ShapeEntropyFunction->SetShapeMatrix(m_ShapeMatrix);
55 
56  // Now allocate an optimizer and set iteration callback.
57  m_Optimizer = OptimizerType::New();
58  m_IterateCallback = itk::MemberCommand<PSMEntropyModelFilter<TImage, TShapeMatrix> >::New();
59  m_IterateCallback->SetCallbackFunction(this, &PSMEntropyModelFilter<TImage, TShapeMatrix>::OptimizerIterateCallback);
60  m_Optimizer->AddObserver(itk::IterationEvent(), m_IterateCallback);
61 
62  // Finally, we need a cost function that combines the
63  // ParticleEntropy function with the ShapeEntropy function.
64  m_CostFunction = PSMTwoCostFunction<Dimension>::New();
65  m_CostFunction->SetFunctionA(m_ParticleEntropyFunction);
66  m_CostFunction->SetFunctionB(m_ShapeEntropyFunction);
67 
68  m_Initialized = false;
69 
70  // Register the Shape Matrix as an attribute of the particle system.
71  // This ensures that the Shape Matrix will receive any particle
72  // AddPosition/RemovePosition/UpdatePosition events, so that it can
73  // update its data accordingly.
74  m_ParticleSystem->RegisterAttribute(m_ShapeMatrix);
75 }

104 template <class TImage, class TShapeMatrix>
105 void
106 PSMEntropyModelFilter<TImage, TShapeMatrix>::AllocateDomainsAndNeighborhoods()
107 {
108  // Allocate all the necessary domains and neighborhoods. This must be done
109  // *after* registering the attributes to the particle system since some of
110  // them respond to AddDomain.
111  for (unsigned int i = 0; i < this->GetNumberOfInputs(); i++)
112  {
113  m_DomainList.push_back( PSMImplicitSurfaceDomain<typename
114  ImageType::PixelType, Dimension>::New() );
115  m_NeighborhoodList.push_back( PSMSurfaceNeighborhood<ImageType>::New() );
116  m_DomainList[i]->SetSigma(m_WorkingImages[i]->GetSpacing()[0] * 2.0);
117  m_DomainList[i]->SetImage(m_WorkingImages[i]);
118 
119  m_ParticleSystem->AddDomain(m_DomainList[i]);
120  m_ParticleSystem->SetNeighborhood(i, m_NeighborhoodList[i]);
121  }
122 }

124 template <class TImage, class TShapeMatrix>
125 void
126 PSMEntropyModelFilter<TImage, TShapeMatrix>::InitializeCorrespondences()
127 {
128  // If the user has specified correspondence points on the input,
129  // then make sure the number of lists of correspondence points
130  // matches the number of inputs (distance transforms). Everything
131  // should match this->GetNumberOfInputs().
132  if (m_InputCorrespondencePoints.size() > 0)
133  {
134  if (this->GetNumberOfInputs() != m_InputCorrespondencePoints.size() )
135  {
136  itkExceptionMacro("The number of inputs does not match the number of correspondence point lists.");
137  }
138  else
139  {
140  for (unsigned int i = 0; i < m_InputCorrespondencePoints.size(); i++)
141  {
142  this->GetParticleSystem()->AddPositionList(m_InputCorrespondencePoints[i],i);
143  }
144  }
145  }
146  else // No input correspondences are specified, so add a point to each surface.
147  {
148  this->CreateSingleCorrespondence();
149  }
150 
151  // Push position information out to all observers (necessary to
152  // correctly fill out the shape matrix)
153  this->GetParticleSystem()->SynchronizePositions();
154 }

156 template <class TImage, class TShapeMatrix>
157 void
158 PSMEntropyModelFilter<TImage, TShapeMatrix>::GenerateData()
159 {
160  // First, undo the default InPlaceImageFilter functionality that will
161  // release our inputs.
162  this->SetInPlace(false);
163 
164  // If this is the first call to GenerateData, then we need to allocate some structures
165  if (this->GetInitialized() == false)
166  {
167  this->AllocateWorkingImages();
168  this->AllocateDataCaches();
169  this->GetOptimizer()->SetCostFunction(m_CostFunction);
170 
171  this->AllocateDomainsAndNeighborhoods();
172 
173  // Point the optimizer to the particle system.
174  this->GetOptimizer()->SetParticleSystem(this->GetParticleSystem());
175  // this->ReadTransforms();
176  this->InitializeCorrespondences();
177  this->InitializeOptimizationFunctions();
178 
179  this->SetInitialized(true);
180  }
181 
182  // This filter can be run in "initialization mode", which causes all
183  // data structures to allocate and initialize, but does not run the
184  // actual optimization.
185  if (this->GetInitializing() == true) return;
186 
187  // Multiscale optimization. This can be stopped and restarted by
188  // calling GenerateData again by maintaining m_CurrentScale.
189  for (unsigned int scale = m_CurrentScale; scale < m_NumberOfScales; scale++)
190  {
191  m_CurrentScale = scale;
192 
193  // Set up the optimization parameters for this scale, unless
194  // this is a restart from a previous call to GenerateData. This
195  // section deals with convergence criteria.
196  if (this->GetAbortGenerateData() == false)
197  {
198  // This section deals with convergence.
199  m_Optimizer->SetTolerance(m_Tolerances[scale]);
200  m_Optimizer->SetMaximumNumberOfIterations(m_MaximumNumberOfIterations[scale]);
201 
202  // Set up the exponentially-decreasing regularlization constant. If
203  // the decay span is greater than 1 iteration, then we will set up
204  // the annealing approach. Otherwise, the optimizer will simply use
205  // its constant annealing parameter.
206  if (m_RegularizationDecaySpan[scale] >= 1.0f)
207  {
208  m_ShapeEntropyFunction->SetMinimumVarianceDecay(m_RegularizationInitial[scale],
209  m_RegularizationFinal[scale],
210  m_RegularizationDecaySpan[scale]);
211  }
212  else
213  {
214  m_ShapeEntropyFunction->SetMinimumVariance(m_RegularizationInitial[scale]);
215  m_ShapeEntropyFunction->SetHoldMinimumVariance(true);
216  }
217  }
218 
219  this->SetAbortGenerateData(false); // reset the abort flag
220 
221  // Finally, run the optimization. Abort flag is checked in the
222  // iteration callback registered with the optimizer. See
223  // this->OptimizerIterateCallback
224  this->GetOptimizer()->StartOptimization();
225 
226  // If this is multiscale, split particles for the next iteration
227  // -- but not if this is the last iteration.
228  if ((m_NumberOfScales > 1) && (scale != m_NumberOfScales-1) )
229  {
230  m_ParticleSystem->SplitAllParticles(this->GetInput()->GetSpacing()[0] * 0.5);
231  }
232 
233  }
234 }

236 template <class TImage, class TShapeMatrix>
237 void
238 PSMEntropyModelFilter<TImage, TShapeMatrix>::InitializeOptimizationFunctions()
239 {
240  // Set the minimum neighborhood radius and maximum sigma based on
241  // the domain of the 1st input image.
242  unsigned int maxdim = 0;
243  double maxradius = 0.0;
244  double spacing = this->GetInput()->GetSpacing()[0];
245  for (unsigned int i = 0; i < TImage::ImageDimension; i++)
246  {
247  if (this->GetInput()->GetRequestedRegion().GetSize()[i] > maxdim)
248  {
249  maxdim = this->GetInput()->GetRequestedRegion().GetSize()[i];
250  maxradius = maxdim * this->GetInput()->GetSpacing()[i];
251  }
252  }
253 
254  // Initialize member variables of the optimization functions.
255  // m_ParticleEntropyFunction->SetMinimumNeighborhoodRadius(maxradius / 3.0);
256  m_ParticleEntropyFunction->SetMinimumNeighborhoodRadius(spacing * 5.0);
257  m_ParticleEntropyFunction->SetMaximumNeighborhoodRadius(maxradius);
258 
259  m_ShapeMatrix->Initialize();
260 }

281 template <class TImage, class TShapeMatrix>
282 bool
283 PSMEntropyModelFilter<TImage, TShapeMatrix>
284 ::CreateSingleCorrespondence()
285 {
286  bool ok = true;
287  for (unsigned int i = 0; i < m_ParticleSystem->GetNumberOfDomains(); i++)
288  {
289  typename itk::ZeroCrossingImageFilter<ImageType, ImageType>::Pointer zc =
290  itk::ZeroCrossingImageFilter<ImageType, ImageType>::New();
291 
292  zc->SetInput( dynamic_cast<PSMImplicitSurfaceDomain<typename ImageType::PixelType, Dimension> *>(
293  m_ParticleSystem->GetDomain(i))->GetImage());
294  zc->Update();
295  itk::ImageRegionConstIteratorWithIndex<ImageType> it(zc->GetOutput(),
296  zc->GetOutput()->GetRequestedRegion());
297 
298  bool done = false;
299  for (it.GoToReverseBegin(); !it.IsAtReverseEnd() && done == false; --it)
300  {
301  if (it.Get() == 1.0)
302  {
303  PointType pos;
304  dynamic_cast<PSMImplicitSurfaceDomain<typename ImageType::PixelType, Dimension> *>(
305  m_ParticleSystem->GetDomain(i))->GetImage()->TransformIndexToPhysicalPoint(it.GetIndex(), pos);
306  try
307  {
308  m_ParticleSystem->AddPosition(pos, i);
309  done = true;
310  }
311  catch(itk::ExceptionObject &)
312  {
313  done = false;
314  ok = false;
315  }
316  }
317  }
318  }
319  return ok;
320 }
