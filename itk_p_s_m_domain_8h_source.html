81  virtual double Distance(const PointType &a, const PointType &b) const
82  {
83  double sum = 0.0;
84  for (unsigned int i = 0; i < VDimension; i++)
85  {
86  sum += (b[i]-a[i]) * (b[i]-a[i]);
87  }
88  return sqrt(sum);
89  }
